custom hooks

              
                import { useState, useEffect } from 'react';

                const useCounter = (initialValue) => {
                  const [count, setCounter] = useState(initialValue)
                  useEffect(() => {
                    console.log('Count is changed')
                    return () => {
                      console.log('Final count is: ' + count)
                    }
                  }, [count])

                  const up = () => {
                    setCounter((state) => state + 1)
                  }

                  const down = () => {
                    setCounter(state => state - 1)
                  }

                  return { up, down }
                }

                export { useCounter }

Strategie comuni

Come gestire una lista di elementi

              
                const elements = ['el-1', 'el-2', 'el-3'];
                
                const MyList = () => {
                  const [disabled, setDisabled] = useState(false);
                  return (
                    <ul>
                      { elements.map((text, i) => (<li key={i}>{ text }</li>)) }
                    </ul>
                  )
                }

Come gestire una chiamata http - basic

              
                import User from 'user.js';
                
                const App = ({ clickHandler, text }) => {
                  const [users, setUsers] = useState([]);
                  useEffect(() => {
                    fetch('/api/users')
                      .then(res => res.json())
                      .then((data) => setUsers(data))
                  }, [])

                  return (
                    <ul>
                      { users.map((user) => (<User data={user} key={user.id} />)) }
                    </ul>
                  )
                }
