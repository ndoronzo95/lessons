Classi e funzioni

              
                class Pokemon {
                  constructor(name, power, lifepoints) {
                    this.name = name;
                    this.power = power;
                    this.lifepoints = lifepoints;
                  }

                  attack(pokemon) {
                    pokemon.lifepoints -= this.power;
                  }
                }

                const pickachu = new Pokemon('Pickachu', 100, 500);
                const charmender = new Pokemon('Charmender', 200, 300);
                pickachu.attack(charmender);
                console.log(charmender.lifepoints) // 200

"Inheritance"

              
                class Rattata extends Pokemon {
                  constructor(lvl) {
                    super('Rattata', 100 * lvl, 300 * lvl)
                  }
                }
                const rattata = new Rattata(10);
                console.log(rattata.power) // 1000
