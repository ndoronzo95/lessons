Dichiaratività (vs imperatività)

              
                function App() {
                  const [content, setContent] = useState('Initial value');
  
                  return (
                    <>
                      <button onClick={() => setContent('New value')}>
                        Change value
                      </button>
                      <p>{ content }</p>
                    <>
                  )
                }
              
            
              
                @Component({
                  template: `
                    <button (click)="content = 'New value'">
                      Change value
                    </button>
                    <p>{{ content }}</p>
                  `
                })
                class AppComponent {
                  content = 'Initial value'
                }
              
            
              
                const button = document.createElement('button');
                button.textContent = 'Change value'

                const p = document.createElement('p');
                p.textContent = 'Initial value';
                
                button.addEventListener('click', () => {
                  p.textContent = 'New value'
                })

                document.body.append(button)
                document.body.append(p)
