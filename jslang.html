                if (actionType === 'eat') {
                  console.log('Having a meal!')
                } else if (actionType === 'drink') {

                  console.log("Chilling...");
                }

switch

              
                const actionType = 'drink'
                switch(actionType) {
                  case 'eat':
                    console.log('Having a meal!')
                    break;
                  case 'drink':
                    console.log("Having a drink!");
                    break;
                  default:
                    console.log("Chilling...")
                }
Notare il break (non obbligatorio)
